CREATE TABLE AUTHORS ( ID INTEGER NOT NULL, FIRST_NAME VARCHAR(25), LAST_NAME VARCHAR(25), PRIMARY KEY (ID));

CREATE TABLE BOOKS ( ISBN VARCHAR(25) NOT NULL, PUBLISHER VARCHAR(25), TITLE VARCHAR(25), PUBLISH_YEAR INTEGER, PRIMARY KEY (ISBN));

CREATE TABLE AUTHORS_BOOKS (ID INTEGER NOT NULL REFERENCES AUTHORS(ID), ISBN VARCHAR(25) NOT NULL REFERENCES BOOKS(ISBN), PRIMARY KEY (ID,ISBN));
